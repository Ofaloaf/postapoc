cb_pro_imperial = {
	name = CB_NAME_PRO_IMPERIAL
	war_name = WAR_NAME_PRO_IMPERIAL
	sprite = 6
	truce_days = 3650
	attacker_can_call_allies = no
	can_ask_to_join_war = no
	major_revolt = yes
	
	can_use = {
		ROOT = {
			vassal_of = FROM	
		}
	}

	is_valid_title = {
		NOT = {
			top_liege = { has_landed_title = e_california }
		}
		OR = {
			tier = emperor
			tier = king
			is_primary_holder_title = yes
		}
	}
	
	on_success = {
		FROM = {
			hidden_tooltip = { disband_event_forces = faction_loyalists }
			set_defacto_liege = e_california
		}
		any_attacker = {
			limit = { character = ROOT }
			participation_scaled_prestige = 100
			hidden_tooltip = {
				disband_event_forces = faction_revolters
			}
		}
		any_attacker = {
			limit = { NOT = { character = ROOT } }
			hidden_tooltip = {
				participation_scaled_prestige = 100 
				disband_event_forces = faction_revolters
			}
		}
		e_california = {
			holder_scope = {
				hidden_tooltip = { letter_event = { id = cetic.210 } }
				reverse_imprison = FROM
			}
		}
	}

	#on_success_title = {
	#	succession_w_cooldown = seniority
	#}

	on_fail = {
		ROOT = {
			prestige = -100
		}
		any_defender = {
			limit = { character = FROM }
			participation_scaled_prestige = 50
		}
		any_defender = {
			limit = { NOT = { character = FROM } }
			hidden_tooltip = { participation_scaled_prestige = 50 }
		}
		
		hidden_tooltip = {
			any_attacker = {
				disband_event_forces = faction_revolters
				add_character_modifier = {
					name = faction_succ_seniority_ultimatum_timer
					hidden = yes
					duration = 3650
				}
			}
		}
		FROM = { hidden_tooltip = { disband_event_forces = faction_loyalists } }
	}

	on_reverse_demand = {
		ROOT = {
			prestige = -200
			prisoner = FROM
		}
		any_defender = {
			limit = { character = FROM }
			participation_scaled_prestige = 100
		}
		any_defender = {
			limit = { NOT = { character = FROM } }
			hidden_tooltip = { participation_scaled_prestige = 100 }
		}
		
		hidden_tooltip = {
			any_attacker = {
				limit = { liege = { character = FROM } }
				prisoner = FROM
				disband_event_forces = faction_revolters
				add_character_modifier = {
					name = faction_pro_imperial_ultimatum_timer
					hidden = yes
					duration = 3650
				}
			}
		}
		
		FROM = { hidden_tooltip = { disband_event_forces = faction_loyalists } }
	}

	attacker_ai_victory_worth = {
		factor = -1 # always accept
	}
	
	attacker_ai_defeat_worth = {
		factor = 100
	}

	defender_ai_victory_worth = {
		factor = -1 # always accept
	}
	
	defender_ai_defeat_worth = {
		factor = 100
	}
	
	is_valid = {
		e_california = { holder_scope = { trait = figurehead } }
		ROOT = {
			NOT = { top_liege = { has_landed_title = e_california } }
			OR = {
				liege = {
					character = PREV # either independent
				}
				liege = { 
					FROM = { 
						is_liege_or_above = PREV # or have shared liege
					}
				}
			}
		}
	}
}

colonization = {
	name = CB_NAME_COLONIZATION
	war_name = WAR_NAME_COLONIZATION
	sprite = 9
	truce_days = 3650
	is_permanent = yes
	hostile_against_others = yes
	check_de_jure_tier = DUKE # this scans all dejure duchies for the counties which are held by or vassals(or below) of selected character. Only valid if is_permanent = yes

	can_use = {
		ROOT = {
			has_character_flag = colonist
		}
	}
	
	can_use_title = {
		FROM = {
			any_realm_province = {
				port = yes
				de_jure_liege_or_above = PREVPREV
				any_province_holding = { holding_type = city }
			}
		}
	}

	#is_valid = {
	#	always = yes
	#}
	
	#is_valid_title = {
	#	always = yes
	#}
	
	on_success = {
		ROOT = { 
			prestige = 250
			hidden_tooltip = {
				clr_character_flag = colonist
				#give_nickname = nick_the_conqueror
				if = {
					limit = { has_landed_title = e_edokko }
					narrative_event = { id = col.12 days = 1 } # Victorious
				}
				if = {
					limit = { has_landed_title = e_novorusskiy }
					narrative_event = { id = col.22 days = 1 } # Victorious
				}
			}
		}
		FROM = {
			prestige = -250 
		}
	}

	on_success_title = {
		custom_tooltip = {
			text = tribal_invasion_succ_tip
			hidden_tooltip = {
				FROM = {
					random_realm_province = {
						limit = {
							port = yes
							de_jure_liege_or_above = PREVPREV
							any_province_holding = { holding_type = city }
							#OR = {
							#	NOT = { culture = ROOT }
							#	NOT = { religion = ROOT }
							#}
						}
						culture = ROOT
						religion = ROOT
						
						# Make it a merchant republic
						random_province_holding = {
							limit = { holding_type = city }
							province_capital = yes
							grant_title = ROOT
						}
						county = { grant_title = ROOT }
						duchy = { grant_title = ROOT }
						# Give them the old capital too
						any_province_holding = {
							limit = { same_realm = FROM }
							grant_title = ROOT
						}
					}
				}
				ROOT = {
					set_government_type = colonial_government
					occupy_minors_of_occupied_settlements = FROM
					gain_all_occupied_titles = FROM
					vassalize_or_take_under_title_destroy_duchies = {
						title = PREV
						enemy = FROM
						type = invasion
					}
					
					# Settle the colonists
					any_demesne_title = {
						limit = { tier = BARON }
						remove_holding_modifier = recently_conquered
						refill_holding_levy = yes
					}
				}
			}
		}
		
	}

	on_fail = {
		FROM = {
			prestige = 100
		}
		ROOT = {
			imprison = FROM 
			hidden_tooltip = {
				disband_event_forces = yes
				clr_character_flag = colonist
			}
		}
	
		ROOT = { prestige = -100 }
		
		random = {
			chance = 40
			if = {
				limit = { ROOT = { has_landed_title = e_edokko } }
				narrative_event = { id = col.13 days = 1 } # Defeated
			}
			if = {
				limit = { has_landed_title = e_novorusskiy }
				narrative_event = { id = col.23 days = 1 } # Defeated
			}
		}
		if = {
			limit = { ROOT = { has_landed_title = e_edokko } }
			activate_title = { title = e_edokko status = no }
		}
		if = {
			limit = { ROOT = { has_landed_title = e_novorusskiy } }
			activate_title = { title = e_novorusskiy status = no }
		}
	}

	on_reverse_demand = {
		ROOT = {
			transfer_scaled_wealth = {
				to = FROM
				value = 4.0
			}
		}
		
		FROM = {
			prestige = 200
		}
		
		ROOT = {
			imprison = FROM
			prestige = -200
			hidden_tooltip = {
				disband_event_forces = yes
				clr_character_flag = colonist
			}
		}
		
		random = {
			chance = 40
			if = {
				limit = { ROOT = { has_landed_title = e_edokko } }
				narrative_event = { id = col.13 days = 1 } # Defeated
			}
			if = {
				limit = { ROOT = { has_landed_title = e_novorusskiy } }
				narrative_event = { id = col.23 days = 1 } # Defeated
			}
		}
		if = {
			limit = { ROOT = { has_landed_title = e_edokko } }
			activate_title = { title = e_edokko status = no }
		}
		if = {
			limit = { ROOT = { has_landed_title = e_novorusskiy } }
			activate_title = { title = e_novorusskiy status = no }
		}
	}
	
	on_attacker_leader_death = {
		ROOT = {
			hidden_tooltip = {
				disband_event_forces = yes
				clr_character_flag = colonist
			}
		}
		end_war = invalid
		
		random = {
			chance = 40
			if = {
				limit = { ROOT = { has_landed_title = e_edokko } }
				narrative_event = { id = col.13 days = 1 } # Defeated
			}
			if = {
				limit = { ROOT = { has_landed_title = e_novorusskiy } }
				narrative_event = { id = col.23 days = 1 } # Defeated

		}
		if = {

	war_name = WAR_NAME_CONSUMERIST_RISING
	sprite = 8
	truce_days = 365
	hostile_against_others = no
	is_permanent = yes
	check_de_jure_tier = KING # this scans all de jure kingdoms for the counties which are held by or vassals(or below) of selected character. Only valid if is_permanent = yes
	apply_short_occ_mod = no # Do not apply the 'recently_conquered' modifier to Holdings

	can_use = {
		ROOT = {
			has_character_flag = consumerist_prophet
		}
	}
	
	is_valid_title = {
		ROOT = {
			has_character_flag = consumerist_prophet
		}
		FROM = {
			any_realm_title = {
				de_jure_liege_or_above = PREVPREV
			}
		}
		d_consumerist = {
			has_holder = no
		}
	}

	on_success = {
		clr_global_flag = consumerist_revolt_ongoing
		set_global_flag = consumerist_revolt_successful
		ROOT = {
			piety = 500
			religion_authority = {
				modifier = rise_of_consumerism
			}
			clr_character_flag = consumerist_prophet
		}
		FROM = {
			piety = -500
		}
	}
	
	on_success_title = {
		custom_tooltip = {
			text = tribal_invasion_succ_tip
			hidden_tooltip = {
				d_consumerist = {
					usurp_title = { target = ROOT type = invasion }
				}
				FROM = {
					random_realm_province = {
						limit = {
							de_jure_liege_or_above = PREVPREV
							NOT = { religion = ROOT }
						}
						religion = ROOT
					}
					random_realm_province = {
						limit = {
							de_jure_liege_or_above = PREVPREV
							NOT = { religion = ROOT }
						}
						religion = ROOT
					}
					random_realm_province = {
						limit = {
							de_jure_liege_or_above = PREVPREV
							NOT = { religion = ROOT }
						}
						religion = ROOT
					}
					random_realm_province = {
						limit = {
							de_jure_liege_or_above = PREVPREV
							NOT = { religion = ROOT }
						}
						religion = ROOT
					}
					random_realm_province = {
						limit = {
							de_jure_liege_or_above = PREVPREV
							NOT = { religion = ROOT }
						}
						religion = ROOT
					}
				}
				ROOT = {
					occupy_minors_of_occupied_settlements = FROM
					gain_all_occupied_titles = FROM
					vassalize_or_take_under_title = {
						title = PREV
						enemy = FROM
					}
					
					any_demesne_title = {
						limit = { tier = BARON }
						remove_holding_modifier = recently_conquered
						refill_holding_levy = yes
					}
					usurp_title = { target = PREV type = invasion }
				}
				
				if = {
					limit = {
						FROM = { is_liege_or_above = ROOT }
					}
					FROM = { imprison = ROOT }
				}
			}
		}
	}

	on_fail = {
		clr_global_flag = consumerist_revolt_ongoing
		FROM = {
			piety = 250
		}
		ROOT = {
			piety = -250
			clr_character_flag = consumerist_prophet
		}
	}

	on_reverse_demand = {
		clr_global_flag = consumerist_revolt_ongoing
		ROOT = {
			piety = -500
			clr_character_flag = consumerist_prophet
			clr_character_flag = the_prophet
			imprison = FROM
		}
		FROM = {
			piety = 250
		}

			limit = { character = FROM }
			participation_scaled_piety = 250
		}

			limit = { NOT = { character = FROM } }
			hidden_tooltip = { participation_scaled_piety = 250 }
		}

	}
	
	on_attacker_leader_death = {

		ROOT = {
			clr_character_flag = consumerist_prophet
			clr_character_flag = the_prophet
		}
		end_war = invalid
		activate_title = { title = d_consumerist status = no }

	}
	
	attacker_ai_defeat_worth = {
		factor = 100
	}

	}

		factor = 100
	}
