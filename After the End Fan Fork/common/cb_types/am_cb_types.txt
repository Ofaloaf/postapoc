		FROM = {
			prestige = -250 
		}

		custom_tooltip = {
			text = tribal_invasion_succ_tip

				FROM = {
					random_realm_province = {
						limit = {
							port = yes
							de_jure_liege_or_above = PREVPREV

		}

	name = CB_NAME_RECONQUISTA_MEXICOAST
	war_name = WAR_NAME_RECONQUISTA_MEXICOAST
	diplo_view_region = custom_mexicoast
	sprite = 9
	truce_days = 3650
	is_permanent = yes
	is_holy_war = yes
	battle_warscore_mult = 0.5
	#max_defender_occupation_score = 85
	allow_whitepeace = no

	allowed_to_target_tributaries = yes
	can_ask_to_join_war = yes

	sort_priority = 10000

	can_use_gui = {
		ROOT = {
			prisoner = no
			is_adult = yes
			OR = {
				NOT = { has_character_modifier = holy_war_cooldown }
				has_game_rule = { name = holy_war_cooldown_rule value = off }
			}
			custom_tooltip = {
				text = RECONQUISTAS_STARTED
				has_global_flag = mexican_reconquistas_unlocked
			}
		}
	}

	can_use = {
		ROOT = {
			has_character_flag = currently_reconquering_mexico
			not_reconquista_check_trigger = yes	

			any_realm_province = {
				OR = {
					AND = {
						any_neighbor_province = {
							region = custom_mexicoast
						}
					}
					AND = {
						NOT = {
							any_neighbor_province = {
								region = custom_mexicoast
							}
						}
						NOT = {
							any_neighbor_province = {
								region = custom_empire_mexico
								owner = {
									OR = {
										AND = {
											independent = yes
											NOT = {
												religion_group = cult_of_saints
											}
										}
										top_liege = {
											NOT = {
												religion_group = cult_of_saints
											}
										}
									}
								}
							}
						}
						any_neighbor_province = {
							is_land = yes
							any_neighbor_province = {
								region = custom_mexicoast
							}
						}
					}
					AND = {
						NOR = {
							any_neighbor_province = {
								region = custom_mexicoast
							}
							any_neighbor_province = {
								is_land = yes
								any_neighbor_province = {
									region = custom_mexicoast 
								}
							}
						}
						NOT = {
							any_neighbor_province = {
								is_land = yes
								any_neighbor_province = {
									region = custom_empire_mexico
									owner = {
										OR = {
											AND = {
												independent = yes
												NOT = {
													religion_group = cult_of_saints
												}
											}
											top_liege = {
												NOT = {
													religion_group = cult_of_saints
												}
											}
										}
									}
								}
							}
						}

						any_neighbor_province = {
							is_land = yes
							any_neighbor_province = {
								is_land = yes
								any_neighbor_province = {
									region = custom_mexicoast
								}
							}
						}
					}
				}
			}

			conditional_tooltip = {
				trigger = {
					ai = yes
				}

				NOT = {
					has_character_modifier = reconquista_preparation_duration
				}
			}
		}

		FROM = {
			any_realm_province = {
				region = custom_mexicoast
			}
			NOT = {
				religion_group = ROOT
			}
		}
	}

	is_valid = {
		ROOT = {
			show_scope_change = no
			custom_tooltip = {
				text = reconquista_valid_TT

				has_character_flag = currently_reconquering_mexico
			}
		}
		custom_tooltip = {
			text = reconquista_valid_2_TT

			FROM = {
				NOT = {
					religion_group = ROOT
				}
			}
		}
	}

	on_add = {
		ROOT = {
			if = {
				limit = {
					has_game_rule = { name = holy_war_cooldown_rule value = on }
				}
				custom_tooltip = {
					text = holy_war_cooldown_ctt
					hidden_tooltip = {
						add_character_modifier = {
							name = holy_war_cooldown
							duration = 3650
						}
					}
				}
			}
			character_event = { id = HF.49210 }
		}
		custom_tooltip = { text = add_allies_reconquista_TT }
	}

	on_success = {

		d_olmeca = {
			save_event_target_as = first_duchy_title
		}

		d_veracruz = {
			save_event_target_as = second_duchy_title
		}

		reconquista_win_effect = yes
	}

	on_reverse_demand = {
		reconquista_lose_effect = yes
	}

	on_invalidation = {
		ROOT = {
			character_event = { id = HF.49200 }
		}
	}

	attacker_ai_victory_worth = {
		factor = -1
	}

	attacker_ai_defeat_worth = {
		factor = 100
	}

	defender_ai_victory_worth = {
		factor = -1
	}

	defender_ai_defeat_worth = {
		factor = 100
	}
}
