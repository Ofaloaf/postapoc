## Fourth Crusade Event (Crusade for the Maritimes)
# Start event (the moment the Crusade starts)
character_event = {
    id = HF.49000
    hide_window = yes
    is_triggered_only = yes
    has_dlc = "Holy Fury"
    trigger = {
		religion = ursuline #Won't happen for Catholics or Mormons, for obvious reasons
		uses_new_crusade = yes
        has_game_rule = {
            name = fourth_crusade
            value = on
        }
    	OR = {
    		AND = {
    			random = 90
    			NOT = {
    				k_maritimes = {
    					owner = {
    						realm_size = 25
    					}
    				}
    			}
    		}
    		AND = {
    			random = 85
    			k_maritimes = {
    				owner = {
    					realm_size = 25
    				}
    			}
    			NOT = {
    				k_maritimes = {
    					owner = {
    						realm_size = 35
    					}
    				}
    			}
    		}
    		AND = {
    			random = 80
    			k_maritimes = {
    				owner = {
    					realm_size = 35
    				}
    			}
    			NOT = {
    				k_maritimes = {
    					owner = {
    						realm_size = 45
    					}
    				}
    			}
    		}
    		AND = {
    			random = 75
    			k_maritimes = {
    				owner = {
    					realm_size = 45
    				}
    			}
    		}
    		has_global_flag = qa_fourth_crusade_event
    	}
    	has_global_flag = first_crusade_complete
        NOT = { has_global_flag = fourth_crusade_launched }
    	k_maritimes = {
    		owner = {
    			realm_size = 50
    			OR = {
    				has_landed_title = c_usitukun
    				any_vassal = {
    					has_landed_title = c_usitukun
    				}
    			}
    			NOT = {
    				religion = ROOT
    			}
    			religion_group = ROOT
    		}
    	}
    	d_nova_scotia = {
    		de_jure_liege = k_maritimes
    	}
        e_canada = {
            has_holder = no
        }
    }

    immediate = {
    	k_maritimes = {
    		owner = {
    			save_event_target_as = fourth_crusade_byzantine_emperor
    			narrative_event = { id = HF.49001 days = 30 }
    		}
    	}
    }
}
