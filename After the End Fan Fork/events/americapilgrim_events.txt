character_event = {
	id = america_pilgrim.20

	hide_window = yes
	is_triggered_only = yes
	
	immediate = {
		# 30	city owned by pilgrim
		if = {
			limit = { character = FROM }
			FROM = { character_event = { id = america_pilgrim.30 } }
		}

		# 31	city owned by a lord in the same realm as pilgrim
		if = {
			limit = {
				NOT = { character = FROM }
				NOT = { FROM = { has_opinion_modifier = { who = ROOT modifier = allowed_pilgrim } } }
				same_realm  = FROM
			}
			FROM = { character_event = { id = america_pilgrim.31 } }
		}

		# 32	city owned by Americanist foreign lord
		if = {
			limit = {
				NOT = { character = FROM }
				NOT = { same_realm  = FROM }
				NOT = { FROM = { has_opinion_modifier = { who = ROOT modifier = allowed_pilgrim } } }
				religion = americanist
			}
			FROM = { character_event = { id = america_pilgrim.32 } }
		}

		# 33	city owned by infidel foreign lord
		if = {
			limit = {
				NOT = { character = FROM }
				NOT = { same_realm  = FROM }
				NOT = { religion = americanist }
				NOT = { FROM = { has_opinion_modifier = { who = ROOT modifier = allowed_pilgrim } } }
			}
			FROM = { character_event = { id = america_pilgrim.33 } }
		}
		
		# 34	city owned by another lord who has already granted you access
		if = {
			limit = {
				NOT = { character = FROM }
				FROM = { has_opinion_modifier = { who = ROOT modifier = allowed_pilgrim } }
			}
			FROM = { character_event = { id = america_pilgrim.34 } }
		}
	}
	
	option = { }
}
