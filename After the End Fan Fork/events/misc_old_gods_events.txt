	}

				de_jure_liege = k_hungary
			}
		}

			copy_title_history = k_magyar

		}

		}

					ROOT = {
						capital = PREV 
					}
				}
			}
			k_hungary = {				
				if = {
					limit = {

		}

			limit = {
				culture = hungarian
