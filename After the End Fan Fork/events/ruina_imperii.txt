namespace = ruin

# Fall of generic empire
narrative_event = {
	id = ruin.1
	title = TITLE_collapse
	desc = EVTDESC_collapse
	picture = GFX_evt_ruina_imperii
	
	major = yes	
	only_playable = yes
	is_triggered_only = yes
	show_root = yes
	trigger = {
		independent = yes
		real_tier = emperor
		in_revolt = no
		
		NOR = {
			government = nomadic_government
			government = merchant_republic_government
			government = colonial_government
		}		
		OR = {
			NOT = {
					realm_size = 120
			}
			AND = {
				NOT = {
					realm_size = 100
				}	
				NOT = {
					dynastic_prestige = 1500
				}
			}
		}
		AND = {
			NOT = {
				primary_title = {
					title = e_california
				}		
			}
			NOT = {
				primary_title = {
					title = e_redcoat_dominion
				}		
			}
			NOT = {
				primary_title = {
					title = e_estelaria
				}
			}
			NOR = {
				primary_title = {
					title = e_edokko
				}
				primary_title = {
					title = e_novorusskiy
				}
				primary_title = {
					title = e_china_west_governor
				}
				primary_title = {
					title = offmap_china
				}
			}
		}
		primary_title = {
			is_titular = yes
			temporary = no
		}
	
	}	
	immediate = {
		ROOT = {
			primary_title = {
				destroy_landed_title = yes
			}
			prestige = -500
		}
	}
	option = {
		name = "RUIN_generic"
		trigger = {
			always = yes
		}		
	}
}

#Fall of Caribbean empire
narrative_event = {
	id = ruin.2
	title = TITLE_collapse_CARIBBEAN
	desc = EVTDESC_collapse_CARIBBEAN
	picture = GFX_evt_ruina_imperii	

	major = yes
	only_playable = yes
	hide_from = yes

	trigger = {
		NOT = {
			has_game_rule = {
				name = caribbean_collapse
				value = off
			}
		}
		independent = yes
		real_tier = emperor
		NOT = {
			has_character_flag = empress_portia
		}
		NOT = {
			year = 2710
		}
		OR = {
			AND = {				
				NOT = {
					realm_size = 90
				}
			}
			AND = {
				NOT = {
					completely_controls = d_haiti
				}
				NOT = {
					completely_controls = d_santo_domingo
				}
				NOT = {
					completely_controls = d_el_cibao
				}
				NOT = {
					completely_controls = k_west_indies
				}
				NOT = {
					completely_controls = k_cuba
				}
			}
		}
		NOT = {
			realm_stewardship = 26
		}
		NOT = {
			realm_diplomacy = 26
		}
		primary_title = {
			temporary = no
			title = e_caribbean
		}
	}	
	mean_time_to_happen = {
		months = 25
	}
	immediate = {
		primary_title = {
			destroy_landed_title = yes
		}
		prestige = -500
	}	
	option = {
		name = "RUIN_CARIBBEAN_NORM"
		trigger = {
			NOT = {
				primary_title = {
					OR = {
						region = world_caribbean
						region = world_gulf_coast
						region = world_central_america_caribbean
						region = world_south_america_caribbean
					}
				}
			}
		}
	}
	option = {
		name = "RUIN_CARIBBEAN_OPPORTUNITY"
		trigger = {
			primary_title = {
				OR = {
					region = world_caribbean
					region = world_gulf_coast
					region = world_central_america_caribbean
					region = world_south_america_caribbean
				}
			}
		}
	}
}
