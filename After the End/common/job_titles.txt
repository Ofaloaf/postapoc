# COUNCILLOR TITLES
##########################################

#	attribute: This attribute is added to the ruler's for governance purposes
#	viceroy: This is the Viceroy title for de jure Kingdoms
#	allow: conditions for the character to hold the title
#	gain_effect: fires when the title is granted
#	lose_effect: fires when a title is lost if dismissal_trigger evaluates to false
#	The other fields are all character modifiers.

job_chancellor = {
	is_chancellor = yes
	attribute = diplomacy
	dignity = 0.33
	opinion_effect = 15
	
	monthly_salary = 0.1
	monthly_prestige = 0.015
	
	allow = {
		OR = {
			is_adult = yes
			has_character_flag = special_chancellor
		}
		
		OR = {
			has_law = true_cognatic_succession
			AND = {
				has_law = agnatic_succession
				is_female = no
			}
			AND = {
				has_law = cognatic_succession
				OR = {
					is_female = no
					diplomacy = 20
				}
			}
			AND = {
				is_female = yes
				has_law = enatic_succession
			}
			AND = {
				has_law = enatic_cognatic_succession
				OR = {
					is_female = yes
					diplomacy = 20
				}
			}
		}
		
		prisoner = no
		NOT = { trait = incapable }
	}
	gain_effect = {
	}
	lose_effect = {
		opinion = { who = FROM modifier = opinion_fired_from_council }
	}
	
	action = action_improve_relations
	action = action_fabricate_claims
	action = action_sow_dissent

	lift_fow = yes
}

job_spiritual = {
	is_spiritual = yes
	attribute = learning
	dignity = 0.15
	opinion_effect = 15
	
	monthly_salary = 0.1
	monthly_piety = 0.015
	
	allow = {
		prisoner = no
		liege = {
			religion = ROOT
		}
		OR = {
			is_adult = yes
			has_character_flag = special_spiritual
		}
		OR = {
			has_law = true_cognatic_succession
			AND = {
				has_law = agnatic_succession
				is_female = no
			}
			AND = {
				has_law = cognatic_succession
				OR = {
					is_female = no
					learning = 20
				}
			}
			AND = {
				is_female = yes
				has_law = enatic_succession
			}
			AND = {
				has_law = enatic_cognatic_succession
				OR = {
					is_female = yes
					learning = 20
				}
			}
			religion = ursuline
		}
		NOT = { trait = incapable }
		
		OR = {
			is_theocracy = yes
			AND = {
				is_ruler = no
				has_autocephaly = no
			}
		}
		
		OR = {
			trait = detached_priest
			trait = martial_cleric
			trait = scholarly_theologian
			trait = mastermind_theologian
		}
	}
	gain_effect = {
	}
	lose_effect = {
		opinion = { who = FROM modifier = opinion_fired_from_council }
	}
	
	action = action_inquisition
	action = action_advance_cul_tech
	action = action_improve_rel_relations
	action = action_charity
	action = action_build_zeal # Tribal replacement for 'action_advance_cul_tech' 

	lift_fow = yes
}
