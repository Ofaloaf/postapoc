# Checks if gender laws/culture/religion allows someone to fight
warrior_trigger = {
	OR = {
		AND = {
			is_female = yes
			OR = {
				religion = gaian
				religion = ursuline
			}
		}
		
		has_character_flag = special_marshal
		
		has_law = true_cognatic_succession
		AND = {
			has_law = agnatic_succession
			is_female = no
		}
		AND = {
			has_law = cognatic_succession
			OR = {
				is_female = no
				martial = 20
			}
		}
		AND = {
			has_law = enatic_succession
			is_female = yes
		}
		AND = {
			has_law = enatic_cognatic_succession
			OR = {
				is_female = yes
				martial = 20
			}
		}
		
		trait = adventurer
		trait = viking
		trait = duelist
		trait = berserker
		trait = peasant_leader
	}
}

ok_sr_sympathy_mission_target_trigger = {
	has_secret_religion_trait_trigger = no
	ROOT = {
		NOT = { secret_religion = { target_type = public target = PREV } }
		NOR = {
			AND = {
				true_religion_group = christian
				PREV = { trait = sympathy_christendom }
			}
			AND = {
				true_religion_group = pagan_group
				PREV = { trait = sympathy_pagans }
			}
			AND = {
				true_religion_group = afro_syncretic
				PREV = { trait = sympathy_afro_syncretic }
			}
			AND = {
				true_religion_group = american_native
				PREV = { trait = sympathy_american_native }
			}
			AND = {
				true_religion_group = cult_of_saints
				PREV = { trait = sympathy_cult_of_saints }
			}
			AND = {
				true_religion_group = pacific
				PREV = { trait = sympathy_pacific }
			}
			AND = {
				true_religion_group = latter_day_saints
				PREV = { trait = sympathy_latter_day_saints }
			}
			AND = {
				true_religion_group = mesoamerican_gods
				PREV = { trait = sympathy_mesoamerican_gods }
			}
			AND = {
				true_religion_group = oldworld
				PREV = { trait = sympathy_oldworld }
			}
			AND = {
				true_religion_group = pirate
				PREV = { trait = sympathy_pirate }
			}
		}
	}
	ROOT = { PREV = { is_openly_of_prevs_secret_religion_trigger = no } }
	ROOT = { PREV = { has_sympathy_for_true_religion_of_prev_trigger = no } }
	prisoner = no
	is_adult = yes
	NOT = { trait = incapable }
	is_inaccessible_trigger = no
	NOT = { has_character_modifier = secret_religion_sympathy_timer }
}
