#Irresistable young wench
character_event = {
	id = 76000
	desc = "EVTDESC76000"
	picture = GFX_evt_lovers
	
	min_age = 16
	max_age = 25
	only_men = yes
	only_rulers = yes
	capable_only = yes
	
	is_triggered_only = yes
	
	trigger = {
		is_married = no
		NOT = {
			trait = chaste
		}
		NOT = {
			trait = celibate
		}
		NOT = {
			trait = homosexual
		}
		fertility = 0.6
		NOT = {
			has_lover = yes
		}
	}
	
	weight_multiplier = {
		days = 1
		
		modifier = {
			factor = 2
			trait = lustful
		}
		modifier = {
			factor = 2
			trait = gregarious
		}
		
		modifier = {
			factor = 0.5
			trait = shy
		}
	}
	
	option = {
		name = "EVTOPTA76000"
		piety = -10
		if = {
			limit = {
				NOT = {
					trait = lustful
				}
			}
			random = {
				chance = 50
				
				add_trait = lustful
				hidden_tooltip = {
					character_event = {
						id = 38272 #Notify Lustful
					}
				}
			}
		}
	}
	option = {
		name = "EVTOPTB76000"
		if = {
			limit = {
				religion_group = muslim
			}
			piety = 5
		}
		if = {
			limit = {
				NOT = {
					religion_group = muslim
				}
			}
			add_trait = chaste
		}
	}
}
