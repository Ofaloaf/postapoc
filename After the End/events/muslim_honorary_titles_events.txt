	id = 105010
	desc = "EVTDESC105010"
	picture = GFX_evt_emissary #Placeholder!

	min_age = 16
	prisoner = no
	capable_only = yes

		war = no
		in_command = no
		OR = {
			religion = americanist
			religion = hamiltonian
			religion = jeffersonian
			religion = libertarian
		}
		NOT = {
			has_minor_title = title_chief_justice
		}

			has_character_flag = qadi_suggester

		NOT = {
			has_character_flag = new_qadi_suggester
		}
		liege = {
			NOT = {
				has_character_modifier = suggested_qadi_timer
			}
			NOT = {
				has_character_modifier = suggested_new_qadi_timer
			}
			NOT = {
				character = ROOT
			}
			prisoner = no
			primary_title = {
				OR = {
					tier = king
					tier = emperor
				}
			}
			OR = {
				religion = americanist
				religion = hamiltonian
				religion = jeffersonian
				religion = libertarian
			}
			NOT = {
				any_vassal = {
					has_minor_title = title_chief_justice
				}
			}
			NOT = {
				any_courtier = {
					has_minor_title = title_chief_justice
				}
			}
			OR = {
				any_vassal = {
					age = 16
					OR = {
						religion = americanist
						religion = hamiltonian
						religion = jeffersonian
						religion = libertarian
					}
					#is_female = no
					prisoner = no
					NOT = {

					}
					NOT = {
						character = ROOT
					}
					NOT = {
						any_rival = {
							character = ROOT
						}
					}
					OR = {
						trait = detached_priest
						trait = martial_cleric
						trait = scholarly_theologian
						trait = mastermind_theologian
					}
				}
				any_courtier = {
					age = 16
					OR = {
						religion = americanist
						religion = hamiltonian
						religion = jeffersonian
						religion = libertarian
					}
					#is_female = no
					prisoner = no
					OR = {
						trait = detached_priest
						trait = martial_cleric
						trait = scholarly_theologian
						trait = mastermind_theologian
					}
					NOT = {
						has_minor_title = title_chief_justice
					}
					NOT = {
						any_rival = {
							character = ROOT
						}
					}
				}
				ROOT = {
					#is_female = no
					NOT = {
						has_minor_title = title_chief_justice
					}
					OR = {
						trait = detached_priest
						trait = martial_cleric
						trait = scholarly_theologian
						trait = mastermind_theologian
					}
				}
			}
		}
	}
	
	mean_time_to_happen = {
		months = 250
		
		modifier = {
			factor = 0.5
			trait = diligent
		}
		modifier = {
			factor = 0.5
			trait = ambitious
		}
		modifier = {
			factor = 2
			trait = slothful
		}
		modifier = {
			factor = 2
			trait = content
		}
	}
	
	immediate = {
		set_character_flag = qadi_suggester

			liege = {
				add_character_modifier = {
					name = suggested_qadi_timer
					duration = 730
					hidden = yes

			}
		}
	}

		name = "EVTOPTA105010" #Suggest a courtier

			liege = {

					age = 16

						religion = libertarian
					}
					prisoner = no
					NOT = { character = PREV }
					NOT = { character = ROOT }
					NOT = {
						has_minor_title = title_chief_justice
					}

						trait = detached_priest

				limit = {
					age = 16

						religion = americanist
