	id = SoA.5420
	desc = EVTDESC_SoA_5420
	picture = GFX_evt_scandal
	border = GFX_event_normal_frame_religion

	only_rulers = yes
	prisoner = no
	min_age = 16
	
	trigger = {
		has_dlc = "Sons of Abraham"
		trait = zealous
		trait = wroth
		NOT = { diplomacy = 5 }
		any_realm_character = {
			NOT = { religion_group = ROOT }
		}
	}
	
	mean_time_to_happen = {
		months = 300
	}
	
	option = {
		name = EVTOPTA_SoA_5420
		hidden_tooltip = {
			any_realm_character = {
				limit = {
					NOT = { religion_group = ROOT }
				}
				opinion = {
					modifier = insulted
					who = ROOT
					years = 1
				}
			}
		}
		custom_tooltip = { text = insulted_realm_infidels }
	}
}
